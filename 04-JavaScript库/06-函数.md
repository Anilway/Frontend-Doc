# 函数

```
// 加减乘除

function jia(num1, num2) {
  console.log(num1 + num2);
}

function jian(num1, num2) {
  console.log(num1 - num2);
}

function cheng(num1, num2) {
  console.log(num1 * num2);
}

function chu(num1, num2) {
  console.log(num1 / num2);
}

jia(1, 2);
jian(1, 2);
cheng(1, 2);
chu(5, 0.5);

/*
  作用域 —— 全局与局部
  return
*/

```
