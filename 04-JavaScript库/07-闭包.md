// 闭包 —— 通过母函数获取到子函数的函数作用域
function user(name) {
  var age, sex;
  return {
    getName: function() {
      return name;
    },
    setName: function(newName) {
      name = newName;
    },
    getAge: function() {
      return age;
    },
    setAge: function(newAge) {
      age = newAge;
    },
    getSex: function() {
      return sex;
    },
    setSex: function(newSex) {
      sex = newSex;
    }
  };
}

var whh = user("王花花");
whh.setAge(20);
whh.setSex("女");
var name = whh.getName();
var age = whh.getAge();
var sex = whh.getSex();

console.log(name, age, sex);
